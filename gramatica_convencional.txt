PROGRAM	        -> 	STATEMENT | FUNCLIST
FUNCLIST	    -> 	FUNCDEF FUNCLIST_LINHA
FUNCLIST_LINHA	-> 	FUNCLIST | ε
FUNCDEF	        -> 	'def' idfuncao '(' PARAMLIST ')' '{' STATELIST '}'
PARAMLIST	    -> 	TIPOS id PARAMLIST_LINHA  | ε
PARAMLIST_LINHA	-> 	',' PARAMLIST | ε
STATELIST	    -> 	STATEMENT
STATEMENT	    -> 	VARDECL ‘,’ | ATRIBSTAT ‘,’ | PRINTSTAT ‘,’ | READSTAT ‘,’ | RETURNSTAT ‘,’ | 'break' ‘,’ | ‘,’ | IFSTAT | FORSTAT | '{' STATELIST '}'
VARDECL	        -> 	TIPOS id VARDECL_LINHA
VARDECL_LINHA	-> 	'['  numero ']' VARDECL_LINHA
LVALUE	        -> 	id LVALUE_LINHA
LVALUE_LINHA	-> 	'[' NUMEXPRESSION ']' LVALUE_LINHA  | ε
ATRIBSTAT	    -> 	LVALUE '=' ATRIBSTAT_LINHA
ATRIBSTAT_LINHA	-> 	EXPRESSION | ALLOCEXPRESSION | FUNCCALL | texto 
ALLOCEXPRESSION	-> 	 'new' TIPOS ALLOCEXPRESSION_LINHA
ALLOCEXPRESSION_LINHA	-> 	'[' NUMEXPRESSION ']' ALLOCEXPRESSION_LINHA
FUNCCALL	    -> 	idfuncao '(' PARAMLISTCALL ')'
PARAMLISTCALL	-> 	texto PARAMLISTCALL_LINHA | EXPRESSION PARAMLISTCALL_LINHA | ε
PARAMLISTCALL_LINHA	    -> 	',' PARAMLISTCALL | ε
PRINTSTAT	    -> 	'print' PRINTSTAT_LINHA
PRINTSTAT_LINHA	-> 	texto | EXPRESSION
READSTAT	    -> 	'read' LVALUE
RETURNSTAT	    -> 	'return' RETURNSTAT_LINHA
RETURNSTAT_LINHA-> 	texto | EXPRESSION | ε
IFSTAT	        -> 	'if' '(' EXPRESSION ')' IFSTAT_LINHA
IFSTAT_LINHA	-> 	'else' STATEMENT | ε
FORSTAT	        -> 	'for' '(' ATRIBSTAT ';' EXPRESSION ';' ATRIBSTAT ')'  STATEMENT
EXPRESSION	    -> 	NUMEXPRESSION EXPRESSION_LINHA
EXPRESSION_LINHA-> 	'>' NUMEXPRESSION | '<' NUMEXPRESSION | '>=' NUMEXPRESSION | '<=' NUMEXPRESSION | '==' NUMEXPRESSION | '!=' NUMEXPRESSION | ε
NUMEXPRESSION	-> 	TERM NUMEXPRESSION_LINHA
NUMEXPRESSION_LINHA	-> 	'+' TERM NUMEXPRESSION_LINHA | '-' TERM NUMEXPRESSION_LINHA | ε
TERM	        -> 	UNARYEXPR TERM_LINHA
TERM_LINHA	    -> 	 '*' UNARYEXPR | '/' UNARYEXPR | ε
UNARYEXPR	    -> 	 '+' FACTOR | '-' FACTOR | FACTOR
FACTOR	        -> 	numero | LVALUE  | 'NULL' | '(' NUMEXPRESSION ')'
