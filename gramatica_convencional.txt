PROGRAM	::= 	STATEMENT | λFUNCLIST
FUNCLIST	::= 	FUNCDEF FUNCLIST_LINHA
FUNCLIST_LINHA	::= 	FUNCLIST | λ
FUNCDEF	::= 	'DEF' FUNCAO id '(' PARAMLIST ')' '{' STATELIST '}'
PARAMLIST	::= 	TIPOS id PARAMLIST_LINHA  | λ
PARAMLIST_LINHA	::= 	',' PARAMLIST | λ
STATELIST	::= 	STATEMENT
STATEMENT	::= 	VARDECL ‘,’ | λATRIBSTAT ‘,’ | λPRINTSTAT ‘,’ | λREADSTAT ‘,’ | λRETURNSTAT ‘,’ | λ'BREAK' ‘,’ | λ‘,’ | λIFSTAT | λFORSTAT | λ'{' STATELIST '}'
VARDECL	::= 	TIPOS id VARDECL_LINHA
VARDECL_LINHA	::= 	'['  numero ']' VARDECL_LINHA
LVALUE	::= 	id LVALUE_LINHA
LVALUE_LINHA	::= 	 '[' NUMEXPRESSION ']' LVALUE_LINHA
ATRIBSTAT	::= 	LVALUE '=' ATRIBSTAT_LINHA
ATRIBSTAT_LINHA	::= 	EXPRESSION | λALLOCEXPRESSION | λFUNCCALL | λtexto 
ALLOCEXPRESSION	::= 	 'NEW' TIPOS ALLOCEXPRESSION_LINHA
ALLOCEXPRESSION_LINHA	::= 	'[' NUMEXPRESSION ']' ALLOCEXPRESSION_LINHA
FUNCCALL	::= 	FUNCAO '(' FUNCCALL_LINHA 
FUNCCALL_LINHA	::= 	 PARAMLISTCALL ')' | λ')'
PARAMLISTCALL	::= 	PARAMLISTCALL_LINHA1 | λPARAMLISTCALL_LINHA2 | λPARAMLISTCALL_LINHA3 | λ
PARAMLISTCALL_LINHA1	::= 	id PARAMLISTCALL_LINHA
PARAMLISTCALL_LINHA2	::= 	texto PARAMLISTCALL_LINHA
PARAMLISTCALL_LINHA3	::= 	EXPRESSION PARAMLISTCALL_LINHA
PARAMLISTCALL_LINHA	::= 	',' PARAMLISTCALL | λ
PRINTSTAT	::= 	'PRINT' PRINTSTAT_LINHA
PRINTSTAT_LINHA	::= 	texto | λEXPRESSION
READSTAT	::= 	'READ' LVALUE
RETURNSTAT	::= 	'RETURN' RETURNSTAT_LINHA
RETURNSTAT_LINHA	::= 	texto | λEXPRESSION | λ
IFSTAT	::= 	'IF' '(' EXPRESSION ')' STATEMENT IFSTAT_LINHA
IFSTAT_LINHA	::= 	'ELSE' STATEMENT | λ
FORSTAT	::= 	'FOR' '(' ATRIBSTAT ';' EXPRESSION ';' ATRIBSTAT ')'  STATEMENT
EXPRESSION	::= 	NUMEXPRESSION EXPRESSION_LINHA1 | λNUMEXPRESSION EXPRESSION_LINHA2 | λNUMEXPRESSION EXPRESSION_LINHA3 | λNUMEXPRESSION EXPRESSION_LINHA4 | λNUMEXPRESSION EXPRESSION_LINHA5 | λNUMEXPRESSION EXPRESSION_LINHA6 
EXPRESSION_LINHA1	::= 	 '>' NUMEXPRESSION  | λ
EXPRESSION_LINHA2	::= 	'<'  NUMEXPRESSION  | λ
EXPRESSION_LINHA3	::= 	'>=' NUMEXPRESSION  | λ
EXPRESSION_LINHA4	::= 	'<=' NUMEXPRESSION  | λ
EXPRESSION_LINHA5	::= 	'==' NUMEXPRESSION  | λ
EXPRESSION_LINHA6	::= 	'!=' NUMEXPRESSION  | λ
NUMEXPRESSION	::= 	TERM NUMEXPRESSION_LINHA
NUMEXPRESSION_LINHA	::= 	'+' TERM NUMEXPRESSION_LINHA | λ'-' TERM NUMEXPRESSION_LINHA | λ
TERM	::= 	UNARYEXPR TERM_LINHA
TERM_LINHA	::= 	 '*' UNARYEXPR | λ'/' UNARYEXPR | λ
UNARYEXPR	::= 	 '+' FACTOR | λ'-' FACTOR | λ FACTOR
FACTOR	::= 	numero | λLVALUE  | λ'NULL' | λ'(' NUMEXPRESSION ')'
